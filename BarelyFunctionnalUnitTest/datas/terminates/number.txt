[zero, inc, dec, one, two, x] {
	zero = {}

	inc = [number, resultInc] {
		resultInc([method] { 
			method() 
			number(method)
		})
	}

	dec = [number, current, previous, method, resultDec] {
		current = {}
		previous = {}
		method = {
			previous = current
			inc(current, [r]{ current = r })
		}
		number(method)
		resultDec(previous)
	}
	
	inc(zero, [r]{ one = r })
	inc(one, [r]{ two = r })
	x = two
	two({dec(x, [r]{ x = r })})
}
